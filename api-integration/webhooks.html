Signature Verification

Verify in 3 Lines. Trust in Zero Doubt.

Always verify the HMAC-SHA256 signature before processing any payload. Skipping this exposes you to replay attacks and forged events.

webhook-verify.js 
const crypto = require('crypto');

// Express.js webhook handler
app.post('/webhooks/brillsign', (req, res) => {
  const secret = process.env.BRILLSIGN_WEBHOOK_SECRET;
  const signature = req.headers['x-brillsign-signature'];
  const rawBody = req.body; // Must be raw buffer, not parsed JSON

  const expected = 'sha256=' + crypto
    .createHmac('sha256', secret)
    .update(rawBody)
    .digest('hex');

  if (!crypto.timingSafeEqual(
    Buffer.from(expected),
    Buffer.from(signature)
  )) {
    return res.status(401).json({ error: 'Invalid signature' });
  }

  const event = JSON.parse(rawBody);
  handleEvent(event); // ✓ Safe to process
  res.status(200).json({ received: true });
});
import hmac, hashlib, os
from flask import Flask, request, jsonify

app = Flask(__name__)

# Flask webhook handler
@app.route('/webhooks/brillsign', methods=['POST'])
def webhook():
    secret = os.environ['BRILLSIGN_WEBHOOK_SECRET'].encode()
    signature = request.headers.get('X-BrillSign-Signature', '')
    raw_body = request.data  # Raw bytes, not .json()

    expected = 'sha256=' + hmac.new(
        secret, raw_body, hashlib.sha256
    ).hexdigest()

    if not hmac.compare_digest(expected, signature):
        return jsonify(error='Invalid signature'), 401

    event = request.get_json()
    handle_event(event)  # ✓ Safe to process
    return jsonify(received=True)

// Get raw body and signature header
$rawBody   = file_get_contents('php://input');

import (
    "crypto/hmac"
    "crypto/sha256"
    "encoding/hex"
    "io"
    "net/http"
)

func webhookHandler(w http.ResponseWriter, r *http.Request) {
    secret := []byte(os.Getenv("BRILLSIGN_WEBHOOK_SECRET"))
    body, _ := io.ReadAll(r.Body)
    signature := r.Header.Get("X-BrillSign-Signature")

    mac := hmac.New(sha256.New, secret)
    mac.Write(body)
    expected := "sha256=" + hex.EncodeToString(mac.Sum(nil))

    if !hmac.Equal([]byte(expected), []byte(signature)) {
        http.Error(w, "Unauthorized", http.StatusUnauthorized)
        return
    }

    handleEvent(body) // ✓ Safe to process
    w.WriteHeader(http.StatusOK)
}

What Happens If You Skip This?

Anyone can craft a fake webhook payload and POST it to your endpoint. Without HMAC verification, you'd process forged signing events — potentially triggering critical business logic with attacker-controlled data.
